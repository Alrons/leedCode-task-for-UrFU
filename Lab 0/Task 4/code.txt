class Solution {
    // Задача состоит в том, чтобы убрать все val из nums и вернуть размер итогового массива
    public int removeElement(int[] nums, int val) {

        // хранит второй индекс
        int index = 0;
        for(int i = 0; i < nums.length; i++){
            
            if(val != nums[i]){
                nums[index] = nums[i];
                index++;
            }
        }
        return index;
    }
    
}
