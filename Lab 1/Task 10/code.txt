class Solution {
    public int singleNumber(int[] nums) {
        // побитовый метод.
        // Как работает, наглядный пример: [4, 1, 2, 1, 2]
        // 0 ^ 4 = 000 ^ 100 = 100 (4)
        // 4 ^ 1 = 100 ^ 001 = 101 (5)
        // 5 ^ 2 = 101 ^ 010 = 111 (7)
        // 7 ^ 1 = 111 ^ 001 = 110 (6)
        // 6 ^ 2 = 110 ^ 010 = 100 (4) ← результат!
        int result = 0;
        for (int num : nums) {
            result ^= num;
        }
        return result;
    }
}
