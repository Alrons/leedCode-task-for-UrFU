class Solution {
    // Вход: цены = [7,1,5,3,6,4]
    //Выход: 5
    //Объяснение: Купить на 2-й день (цена = 1) и продать на 5-й день (цена = 6), прибыль = 6-1 = 5.
    //Обратите внимание, что покупка во 2-й день и продажа в 1-й день не допускаются, поскольку сначала нужно купить, а потом продать.
    public int maxProfit(int[] prices) {
        int maxProfit = 0;
        int minPrice = Integer.MAX_VALUE;
        
        for (int i = 0; i < prices.length; i++) {
            if (prices[i] < minPrice) {
                minPrice = prices[i];
            } else if (prices[i] - minPrice > maxProfit) {
                maxProfit = prices[i] - minPrice;
            }
        }
        
        return maxProfit;

    }
}
