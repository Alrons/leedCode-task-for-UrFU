class Solution {
    public int missingNumber(int[] nums) {
        
        int n = nums.length;
        // ожидаемая сумма
        int expectedSum = n * (n + 1) / 2;
        // фактическая сумма
        int actualSum = 0;
        
        for (int num : nums) {
            actualSum += num;
        }
        
        
        return expectedSum - actualSum;
    }
}
