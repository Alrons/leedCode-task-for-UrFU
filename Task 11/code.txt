class Solution {
    public int smallestNumber(int n) {
        // Находим длину числа 
        // пример: length = 32 - numberOfLeadingZeros(5) = 32 - 29 = 3
        int length = 32 - Integer.numberOfLeadingZeros(n);

        // тут пример: candidate = (1 << 3) - 1 = (1000₂ - 1) = 111₂ = 7
        return (1 << length) - 1;
    }
}
