class Solution {

    // Вам дано большое целое число, представленное в виде массива целых чисел digits, где каждая цифра digits[i]— это его цифра. Цифры упорядочены слева направо от старшей к младшей. Большое целое число не содержит начальных символов «».ith0 Увеличьте большое целое число на единицу и верните полученный массив цифр .

    //Прмер: 
    //Ввод: цифры = [1,2,3]
    //Вывод: [1,2,4]
    //Объяснение: Массив представляет целое число 123.
    //Увеличение на единицу дает 123 + 1 = 124.
    //Таким образом, результат должен быть [1,2,4].
    public int[] plusOne(int[] digits) 
    {
        // обрабарываем только те что имеют размер 1 и имеют число 9
        if (digits.length == 1){
            if (digits[digits.length-1] == 9){
                 int[] result = {1, 0};
                 return result;
            }
        }

        // обрабатываем только цифры где 9 стоит везде кроме самого первого элемента
        for(int i = digits.length; i > 0; i--)
        {
            if(digits[i-1] != 9)
            {
                digits[i-1] = digits[i-1] + 1;
                return digits;
            }
            else{
                digits[i-1] = 0;
            }
        }

        // если цыфры 9 стоят везде, то выводим это
        int[] result = new int[digits.length + 1];
        result[0] = 1;
        return result;
    }
}
